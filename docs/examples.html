Examples

Basic Onboarding Tour (Vanilla JS)

import { createGuideFlow } from '@guideflow/core';
import '@guideflow/core/styles';

const gf = createGuideFlow({ theme: 'minimal' });

await gf.start({
  id: 'onboarding',
  initial: 'step-1',
  states: {
    'step-1': {
      steps: [{ id: 'step-1', content: { title: 'Welcome!', body: 'Let us show you around.' }, target: '#header', placement: 'bottom' }],
      on: { NEXT: 'step-2' },
    },
    'step-2': {
      steps: [{ id: 'step-2', content: { title: 'Dashboard', body: 'All your data lives here.' }, target: '#dashboard', placement: 'right' }],
      on: {},
      final: true,
    },
  },
});

React — useTour Hook

import { TourProvider, useTour } from '@guideflow/react';
import { gf } from './guideflow';
import { onboardingFlow } from './flows/onboarding';

function App() {
  return (
    <TourProvider instance={gf}>
      <StartButton />
    </TourProvider>
  );
}

function StartButton() {
  const { start, stop, next, isActive, currentStepIndex, totalSteps } = useTour();
  return (
    <div>
      <button onClick={() => start(onboardingFlow)}>Start Tour</button>
      {isActive && (
        <span>Step {currentStepIndex + 1} / {totalSteps}</span>
      )}
    </div>
  );
}

Vue 3 — useTour Composable

<script setup>
import { useTour } from '@guideflow/vue';
import onboardingFlow from './flows/onboarding';

const tour = useTour();
</script>

<template>
  <button @click="tour.start(onboardingFlow)">Start Tour</button>
  <p v-if="tour.isActive.value">
    Step {{ tour.currentStepIndex.value + 1 }} of {{ tour.totalSteps.value }}
  </p>
</template>

Svelte — createTourStore

<script>
  import { createTourStore } from '@guideflow/svelte';
  import myFlow from './flows/onboarding';

  const tour = createTourStore({ theme: 'bold' });
</script>

<button on:click={() => tour.start(myFlow)}>Start</button>

{#if $tour.isActive}
  <p>Step {$tour.currentStepIndex + 1} of {$tour.totalSteps}</p>
{/if}

AI-Generated Tour

import { createGuideFlow } from '@guideflow/core';
import { createAI, OpenAIProvider } from '@guideflow/ai';

const gf = createAI(
  new OpenAIProvider({ apiKey: import.meta.env.VITE_OPENAI_KEY }),
  createGuideFlow({})
);

// Generate a tour from a natural language prompt
const steps = await gf.ai.generate('Walk the user through creating a new project');
await gf.start({ id: 'ai-tour', steps });

// Answer a user question
const answer = await gf.ai.chat('Where do I find my billing settings?');
console.log(answer.answer);

Analytics + A/B Testing

import { createGuideFlow } from '@guideflow/core';
import { AnalyticsCollector, PostHogTransport, ExperimentEngine } from '@guideflow/analytics';

// A/B test the tour theme
const engine = new ExperimentEngine(currentUser.id);
const { value: theme } = engine.assign({
  id: 'tour-theme',
  variants: [
    { id: 'control', value: 'minimal' },
    { id: 'treatment', value: 'bold' },
  ],
});

const gf = createGuideFlow({ theme });

// Track events
new AnalyticsCollector({ userId: currentUser.id })
  .addTransport(new PostHogTransport())
  .attach(gf);
