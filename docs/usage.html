Usage Guide

1. Create an Instance

import { createGuideFlow } from '@guideflow/core';
import '@guideflow/core/styles';

export const gf = createGuideFlow({ theme: 'minimal' });

2. Define a Flow

Flows are finite-state machines. Each key in states maps to one or more steps.

import type { FlowDefinition } from '@guideflow/core';

export const onboardingFlow: FlowDefinition = {
  id: 'onboarding',
  initial: 'welcome',
  states: {
    welcome: {
      steps: [{
        id: 'welcome',
        content: { title: 'Welcome!', body: 'Let us show you around.' },
        target: '#app-header',
        placement: 'bottom',
      }],
      on: { NEXT: 'dashboard' },
    },
    dashboard: {
      steps: [{
        id: 'dashboard',
        content: { title: 'Your Dashboard', body: 'All your data lives here.' },
        target: '#dashboard',
        placement: 'right',
      }],
      on: {},
      final: true,
    },
  },
};

3. Start the Tour

import { gf } from './guideflow';
import { onboardingFlow } from './flows/onboarding';

await gf.start(onboardingFlow);

4. React Integration

import { TourProvider, useTour } from '@guideflow/react';

function App() {
  return (
    <TourProvider instance={gf}>
      <MyApp />
    </TourProvider>
  );
}

function StartButton() {
  const { start } = useTour();
  return <button onClick={() => start(onboardingFlow)}>Start Tour</button>;
}

5. AI-Generated Tours

import { createAI, OpenAIProvider } from '@guideflow/ai';

const gfWithAI = createAI(
  new OpenAIProvider({ apiKey: import.meta.env.VITE_OPENAI_KEY }),
  gf
);

const steps = await gfWithAI.ai.generate('Walk me through checkout');
await gfWithAI.start({ id: 'ai-tour', steps });

6. Analytics

import { AnalyticsCollector, PostHogTransport } from '@guideflow/analytics';

const collector = new AnalyticsCollector({ userId: currentUser.id })
  .addTransport(new PostHogTransport());

collector.attach(gf);
